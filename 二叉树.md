## 概述

二叉树普遍会用到递归算法，形式类似于

```Go
func alg(root *TreeNode) *TreeNode {
  if root == nil {
    return nil
  }
  // 某种逻辑
  ......
  return alg(root.Left) && alg(root.Right)
}
```

func maxDepth(root *TreeNode) int {
    if root == nil {
        return 0
    }
    l := maxDepth(root.Left)
    r := maxDepth(root.Right)
    return max(l, r) + 1
}

    if p == nil || q == nil {

### 翻转 226

```Go
func invertTree(root *TreeNode) *TreeNode {
    if root == nil {
        return nil
    }
    root.Left, root.Right = root.Right, root.Left
    invertTree(root.Left)
    invertTree(root.Right)
    return root    
}
```

### 路径总和 112

```Go
func hasPathSum(root *TreeNode, targetSum int) bool {
    if root == nil {
        return false
    }
    if root.Val == targetSum && root.Left == nil && root.Right == nil {
        return true
    }
    return hasPathSum(root.Left, targetSum-root.Val) || hasPathSum(root.Right, targetSum-root.Val)
}
```

### 平衡二叉树 110

```Go
func isBalanced(root *TreeNode) bool {
    return height(root) != -1
}

func height(root *TreeNode) int {
    if root == nil {
        return 0
    }
    height_l := height(root.Left)
    if height_l == -1 {
        return -1
    }
    height_r := height(root.Right)
    if height_r == -1 {
        return -1
    }
    if abs(height_l-height_r) > 1 {
        return -1
    }
    return max(height_l, height_r)+1
}

func abs(i int) int {
    if i > 0 {
        return i
    }
    return -i
}
```

### 二叉树的右视图 199

```Go
func rightSideView(root *TreeNode) []int {
  	// dfs + 优先右子节点
    ans := []int{}
    var dfs func(*TreeNode, int)
    dfs = func(node *TreeNode, depth int) {
        if node == nil {
            return 
        }
        if depth == len(ans) {
            ans = append(ans, node.Val)
        }
        dfs(node.Right, depth+1)
        dfs(node.Left, depth+1)
    }
    dfs(root, 0)
    return ans
}
```

### 从前序与中序遍历序列构造二叉树

```Go
func buildTree(preorder []int, inorder []int) *TreeNode {
    n := len(preorder)
    if n == 0 {
        return nil
    }
    leftIndex := slices.Index(inorder, preorder[0])
    left := buildTree(preorder[1:leftIndex+1], inorder[0:leftIndex])
    right := buildTree(preorder[leftIndex+1:], inorder[leftIndex+1:])
    root := &TreeNode{preorder[0], left, right}
    return root 
}
```
